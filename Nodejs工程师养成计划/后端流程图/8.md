    USER {
        id int
        username varchar
        email varchar
        password varchar
        created_at datetime
        
        PK id
        UNIQUE email
    }

    ROLE {
        id int
        name varchar
        
        PK id
        UNIQUE name
    }

    PERMISSION {
        id int
        name varchar
        
        PK id
        UNIQUE name
    }

    ARTICLE {
        id int
        title varchar
        content text
        author_id int
        category_id int
        created_at datetime
        
        PK id
        FK author_id USER.id
        FK category_id CATEGORY.id
    }

    PRODUCT {
        id int
        name varchar
        price float
        category_id int
        created_at datetime
        
        PK id
        FK category_id CATEGORY.id
    }

    CATEGORY {
        id int
        name varchar
        
        PK id
        UNIQUE name
    }

    COMMENT {
        id int
        user_id int
        article_id int
        content text
        created_at datetime
        
        PK id
        FK user_id USER.id
        FK article_id ARTICLE.id
    }

    ORDER {
        id int
        user_id int
        product_id int
        order_date datetime
        total_price float
        
        PK id
        FK user_id USER.id
        FK product_id PRODUCT.id
    }

    FILE {
        id int
        file_path varchar
        file_type varchar
        uploaded_at datetime
        
        PK id
    }

    USER ||--o{ ROLE : "has"
    ROLE ||--o{ PERMISSION : "grants"
    USER ||--o{ ARTICLE : "writes"
    USER ||--o{ COMMENT : "writes"
    ARTICLE ||--|| CATEGORY : "belongs to"
    PRODUCT ||--|| CATEGORY : "belongs to"
    USER ||--o{ ORDER : "places"
    ORDER ||--|| PRODUCT : "contains"
    FILE ||--o{ ARTICLE : "attaches"
